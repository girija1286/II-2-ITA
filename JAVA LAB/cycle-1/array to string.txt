import java.util.*;
import java.lang.*;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc = new Scanner(System.in);
	    System.out.print("enter the size: ");
	    int n = sc.nextInt();
	    char[] a = new char[n];
	    String s = "";
	    System.out.println("enter array characters");
	    for(int i=0;i<n;i++){
	        a[i]=sc.next().charAt(0);
	        s+=a[i];
	    }
	     System.out.println("converted string from array : "+s); 
	}
}
